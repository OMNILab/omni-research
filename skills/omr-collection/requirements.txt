# omr-collection Skill Dependencies
# Optional dependencies for enhanced features with graceful fallback

# ========================================
# CORE DEPENDENCIES (Required)
# ========================================
# These are required for basic functionality
# Install: pip install requests pdfplumber PyPDF2 html2text

requests>=2.28.0           # HTTP client for API calls and web fetching
pdfplumber>=0.7.0          # PDF text extraction (fallback PDF parser)
PyPDF2>=3.0.0              # PDF text extraction (last resort fallback)
html2text>=2020.1.16       # HTML to markdown conversion

# ========================================
# OPTIONAL: arxiv SDK (Enhanced Paper Downloads)
# ========================================
# Recommended for reliable arxiv paper downloads with rich metadata
# Benefits:
#   - Built-in retry logic (more reliable downloads)
#   - Rich metadata: title, authors, DOI, categories, abstract
#   - Progress tracking for large PDFs
#   - Better error handling
# Install: pip install arxiv
# Fallback behavior: HTTP direct download (basic metadata only)

arxiv>=3.0.0               # Official arxiv Python SDK

# ========================================
# OPTIONAL: MCP SDK (Chrome MCP Integration)
# ========================================
# Required for webpage screenshots and search automation
# Benefits:
#   - Rendered webpage screenshots (PNG capture)
#   - Better markdown conversion from rendered pages
#   - Google Scholar search automation
# Install: pip install mcp
# Also requires: Chrome MCP server (npm package)
# Fallback behavior: HTTP fetch (markdown only, no screenshots)

mcp>=1.0.0                 # Anthropic MCP Python SDK

# ========================================
# OPTIONAL: HuggingFace Hub (Dataset/Model Downloads)
# ========================================
# Required for downloading full datasets or model weights
# Benefits:
#   - Full dataset download (not just README)
#   - Model weights download
#   - Proper handling of HF Hub structure
# Install: pip install huggingface_hub
# Fallback behavior: README + card metadata only (no download)

huggingface_hub>=0.20.0    # HuggingFace Hub library

# ========================================
# INSTALLATION GUIDE
# ========================================

# Option 1: Minimal installation (core only)
# pip install requests pdfplumber PyPDF2 html2text

# Option 2: Enhanced paper handling (arxiv SDK)
# pip install -r requirements.txt  # Includes arxiv

# Option 3: Full features (arxiv + MCP)
# pip install -r requirements.txt
# npm install -g @anthropic/chrome-mcp-server  # Chrome MCP server

# Option 4: All features including HF downloads
# pip install -r requirements.txt
# npm install -g @anthropic/chrome-mcp-server
